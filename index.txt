<script>
    // 1. XỬ LÝ LƯỢT TRUY CẬP
    let totalViews = localStorage.getItem('trihieu_views');
    if (!totalViews) { totalViews = 1500; } // Con số khởi tạo cho đẹp
    totalViews = parseInt(totalViews) + 1;
    localStorage.setItem('trihieu_views', totalViews);
    document.getElementById('total-views').innerText = totalViews.toLocaleString();

    // 2. MÔ PHỎNG ĐANG TRỰC TUYẾN
    setInterval(() => {
        const randomOnline = Math.floor(Math.random() * (12 - 3 + 1)) + 3;
        document.getElementById('online-now').innerText = randomOnline;
    }, 4000);

    // 3. XỬ LÝ BÌNH LUẬN
    function loadComments() {
        const comments = JSON.parse(localStorage.getItem('trihieu_comments')) || [];
        const display = document.getElementById('comment-display');
        if (comments.length === 0) {
            display.innerHTML = '<p style="color:#999; text-align:center; padding:10px">Chưa có bình luận nào.</p>';
            return;
        }
        display.innerHTML = comments.map(c => `
            <div class="comment-item">
                <strong>${c.name}</strong>
                <small>${c.date}</small>
                <p style="margin: 5px 0 0 0">${c.text}</p>
            </div>
        `).join('');
    }

    function addComment() {
        const nameInput = document.getElementById('visitor-name');
        const textInput = document.getElementById('visitor-comment');
        
        if (!nameInput.value.trim() || !textInput.value.trim()) {
            alert("Vui lòng nhập tên và nội dung bình luận!");
            return;
        }

        const newComment = {
            name: nameInput.value,
            text: textInput.value,
            date: new Date().toLocaleString('vi-VN')
        };

        const comments = JSON.parse(localStorage.getItem('trihieu_comments')) || [];
        comments.unshift(newComment);
        localStorage.setItem('trihieu_comments', JSON.stringify(comments));

        nameInput.value = "";
        textInput.value = "";
        loadComments();
    }

    window.onload = loadComments;
</script>
